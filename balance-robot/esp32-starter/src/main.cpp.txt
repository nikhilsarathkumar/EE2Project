#include <Arduino.h>
#include <BluetoothSerial.h>
#include <TimerInterrupt_Generic.h>
#include <Adafruit_MPU6050.h>
#include <Adafruit_Sensor.h>
#include <step.h>

// ── Controller gains  (3-state LQR: z = [theta, theta_dot, x_dot]) ───────────
// NOTE: the control input is now wheel ANGULAR ACCELERATION (rad/s^2), not a
// speed. These gains are in rad/s^2 per state and DO NOT transfer from the old
// speed-law values — recompute them from the 3-state LQR (compute_gains.py)
// with the new a_tt. Values below are PLACEHOLDERS so the loop is complete.
const float K_THETA     = -550.0f;   // rad/s^2 per rad
const float K_THETA_DOT =  -80.0f;   // rad/s^2 per (rad/s)
const float K_XDOT      =    0.0f;   // rad/s^2 per (m/s) — the 3-state addition.
                                     // start at 0 to confirm balance, then tune up.

// ── Limits ──────────────────────────────────────────────────────────────────
const float ACCEL_MAX   = 500.0f;    // clamp on commanded accel (rad/s^2).
                                     // matches the stepper ramp so the command
                                     // can't outrun what the motor can deliver.
const float OMEGA_MAX   =  30.0f;    // clamp on integrated commanded speed (rad/s)

// ── Estimator / sensor settings ─────────────────────────────────────────────
const float GYRO_BIAS      = -0.02f;
const float CF_ALPHA       = 0.98f;
const float THETA_OFFSET   = 0.0777f;
const float GYRO_LPF       = 0.95f;
const float COM_HEIGHT_M   = 0.105f;   // CoM height above wheel axle
const float WHEEL_RADIUS_M = 0.034f;   // wheel radius

// ── Pins ──────────────────────────────────────────────────────────────────────
const int STEPPER1_DIR_PIN  = 16;
const int STEPPER1_STEP_PIN = 17;
const int STEPPER2_DIR_PIN  = 4;
const int STEPPER2_STEP_PIN = 14;
const int STEPPER_EN_PIN    = 15;
const int TOGGLE_PIN        = 32;

const int   LOOP_INTERVAL_MS    = 10;
const int   PRINT_INTERVAL_MS   = 20;
const int   STEPPER_INTERVAL_US = 20;
const float DT = LOOP_INTERVAL_MS / 1000.0f;

static bool capActive = false;

// ── Objects ───────────────────────────────────────────────────────────────────
BluetoothSerial BT;
ESP32Timer ITimer(3);
Adafruit_MPU6050 mpu;
step step1(STEPPER_INTERVAL_US, STEPPER1_STEP_PIN, STEPPER1_DIR_PIN);
step step2(STEPPER_INTERVAL_US, STEPPER2_STEP_PIN, STEPPER2_DIR_PIN);

bool TimerHandler(void *)
{
  step1.runStepper();
  step2.runStepper();
  return true;
}

template<typename T> void logPrint(T v)           { Serial.print(v);      if (BT.hasClient()) BT.print(v); }
template<typename T> void logPrint(T v, int dp)   { Serial.print(v, dp);  if (BT.hasClient()) BT.print(v, dp); }
template<typename T> void logPrintln(T v)         { Serial.println(v);    if (BT.hasClient()) BT.println(v); }
template<typename T> void logPrintln(T v, int dp) { Serial.println(v,dp); if (BT.hasClient()) BT.println(v,dp); }


void setup()
{
  Serial.begin(115200);
  BT.begin("BalanceRobot");
  pinMode(TOGGLE_PIN, OUTPUT);

  if (!mpu.begin()) {
    Serial.println("MPU6050 not found");
    while (1) delay(10);
  }

  mpu.setAccelerometerRange(MPU6050_RANGE_2_G);
  mpu.setGyroRange(MPU6050_RANGE_250_DEG);
  mpu.setFilterBandwidth(MPU6050_BAND_10_HZ);

  if (!ITimer.attachInterruptInterval(STEPPER_INTERVAL_US, TimerHandler)) {
    Serial.println("Stepper ISR failed");
    while (1) delay(10);
  }

  step1.setAccelerationRad(500.0f);
  step2.setAccelerationRad(500.0f);

  pinMode(STEPPER_EN_PIN, OUTPUT);
  digitalWrite(STEPPER_EN_PIN, LOW);

  pinMode(0, INPUT_PULLUP);
  Serial.println("Press BOOT to start controller...");
  while (digitalRead(0) == HIGH) delay(10);
  while (digitalRead(0) == LOW)  delay(10);
}

void loop()
{
  static unsigned long loopTimer  = millis();
  static unsigned long printTimer = millis();
  static float theta_filt     = 0.0f;
  static float theta_dot_filt = 0.0f;
  static float vel_filt       = 0.0f;
  static float omega_c        = 0.0f;   // integrated commanded wheel speed (rad/s)

  if (millis() < loopTimer) return;
  loopTimer = millis() + LOOP_INTERVAL_MS;

  sensors_event_t a, g, temp;
  mpu.getEvent(&a, &g, &temp);

  // 3-sample median filter on raw gyro
  static float gyrobuf[3] = {0, 0, 0};
  static uint8_t gyrobuf_i = 0;
  gyrobuf[gyrobuf_i] = g.gyro.y;
  gyrobuf_i = (gyrobuf_i + 1) % 3;
  float ga = gyrobuf[0], gb = gyrobuf[1], gc = gyrobuf[2];
  float gyro_med = max(min(ga, gb), min(max(ga, gb), gc));

  float theta_dot   = gyro_med - GYRO_BIAS;
  float theta_accel = asinf(constrain(a.acceleration.z / 9.81f, -1.0f, 1.0f)) - THETA_OFFSET;

  theta_dot_filt = GYRO_LPF * theta_dot_filt + (1.0f - GYRO_LPF) * theta_dot;
  theta_filt     = CF_ALPHA * (theta_filt + theta_dot_filt * DT) + (1.0f - CF_ALPHA) * theta_accel;

  // ── Body velocity estimate  (x_dot state) ─────────────────────────────────
  // v_body = r*omega_wheel + L*theta_dot*cos(theta)
  float vel_raw = step1.getSpeedRad() * WHEEL_RADIUS_M + COM_HEIGHT_M * theta_dot_filt * cosf(theta_filt);
  vel_filt = GYRO_LPF * vel_filt + (1.0f - GYRO_LPF) * vel_raw;

  // ── 3-state feedback → wheel ANGULAR ACCELERATION command (rad/s^2) ────────
  float u_th  = -K_THETA     * theta_filt;
  float u_thd = -K_THETA_DOT * theta_dot_filt;
  float u_xd  = -K_XDOT      * vel_filt;
  float u     = u_th + u_thd + u_xd;                 // u = omega_c_dot
  u = constrain(u, -ACCEL_MAX, ACCEL_MAX);

  // ── Integrate accel → commanded wheel speed, then command the motors ───────
  omega_c += u * DT;
  omega_c = constrain(omega_c, -OMEGA_MAX, OMEGA_MAX);   // clamp = integral anti-windup

  step1.setTargetSpeedRad(omega_c);
  step2.setTargetSpeedRad(omega_c);   // <- negate this one if your wheels are mirrored

  if (!capActive && millis() >= printTimer) {
    printTimer += PRINT_INTERVAL_MS;
    logPrint("t=");      logPrint(millis());
    logPrint(" th=");    logPrint(theta_filt,     4);
    logPrint(" th_d=");  logPrint(theta_dot_filt, 4);
    logPrint(" td_r=");  logPrint(theta_dot,      4);
    logPrint(" vel=");   logPrint(vel_filt,       4);
    logPrint(" u_th=");  logPrint(u_th,           4);
    logPrint(" u_thd="); logPrint(u_thd,          4);
    logPrint(" u_xd=");  logPrint(u_xd,           4);
    logPrint(" u=");     logPrint(u,              4);   // commanded accel (rad/s^2)
    logPrint(" wc=");    logPrint(omega_c,        4);   // commanded speed  (rad/s)
    logPrint(" spd=");   logPrintln(step1.getSpeedRad(), 4);
  }
}
